// ── Parse CSV ──

function parseCSV(text) {

  const lines = text.trim().split('\n');

  if (lines.length < 2) return [];

  const headers = lines[0].split(',').map(h => h.replace(/^"|"$/g, '').trim());

  return lines.slice(1).map(line => {

    // Handle quoted fields with commas

    const fields = [];

    let cur = '', inQ = false;

    for (let i = 0; i < line.length; i++) {

      if (line[i] === '"') { inQ = !inQ; }

      else if (line[i] === ',' && !inQ) { fields.push(cur.trim()); cur = ''; }

      else { cur += line[i]; }

    }

    fields.push(cur.trim());

    const obj = {};

    headers.forEach((h, i) => obj[h] = (fields[i] || '').replace(/^"|"$/g, '').trim());

    return obj;

  }).filter(r => r.Name && r.Name.length > 0);

}


// ── Format date ──

function parseDate(str) {

  if (!str) return null;

  // Accept DD/MM/YYYY or YYYY-MM-DD or "15 March 2025"

  let d = new Date(str);

  if (!isNaN(d)) return d;

  const parts = str.split('/');

  if (parts.length === 3) return new Date(`${parts[2]}-${parts[1].padStart(2,'0')}-${parts[0].padStart(2,'0')}`);

  return null;

}

// ── Render cards ──

let allEvents = [];

let activeFilter = 'all';

let showPast = false;


function renderEvents() {

  const grid = document.getElementById('events-grid');

  const today = new Date(); today.setHours(0,0,0,0);


  let filtered = allEvents.filter(e => {

    const typeMatch = activeFilter === 'all' || (e.Type || '').toLowerCase().includes(activeFilter);

    return typeMatch;

  });


  const upcoming = filtered.filter(e => {

    const d = parseDate(e.Date);

    return !d || d >= today;

  });


  const past = filtered.filter(e => {

    const d = parseDate(e.Date);

    return d && d < today;

  });


  if (filtered.length === 0) {

    grid.innerHTML = `<div class="state-msg">

      <div class="icon">📭</div>

      <p>No events found for this filter.</p>

    </div>`;

    return;

  }


  let html = '';


  if (upcoming.length === 0) {

    html += `<div class="state-msg" style="grid-column:1/-1">

      <div class="icon">🌙</div>

      <p>No upcoming events right now — check back soon insha'Allah.</p>

    </div>`;

  } else {

    upcoming.forEach((e, i) => { html += buildCard(e, i); });

  }


  if (past.length > 0) {

    if (!showPast) {

      html += `<div class="past-notice" style="grid-column:1/-1">

        <a onclick="togglePast()">Show ${past.length} past event${past.length > 1 ? 's' : ''}</a>

      </div>`;

    } else {

      html += `<div class="section-divider" style="grid-column:1/-1"><span>Past Events</span></div>`;

      past.forEach((e, i) => { html += buildCard(e, i, true); });

      html += `<div class="past-notice" style="grid-column:1/-1">

        <a onclick="togglePast()">Hide past events</a>

      </div>`;

    }

  }


  grid.innerHTML = html;

}


function buildCard(e, i, isPast = false) {

  const date = parseDate(e.Date);

  const dateChip = date ? `

    <div class="event-date-chip">

      <span class="day">${date.getDate()}</span>

      <span class="month">${MONTHS[date.getMonth()]}</span>

    </div>` : '';


  const imgSection = e.ImageURL

    ? `<img src="${e.ImageURL}" alt="${e.Name}" onerror="this.parentElement.innerHTML='<div class=event-img-placeholder>${tc.emoji}</div>'">`

    : `<div class="event-img-placeholder">${tc.emoji}</div>`;


  const signupBtn = e.FormLink

    ? `<a class="event-signup" href="${e.FormLink}" target="_blank">Sign Up ↗</a>`

    : `<span class="event-signup no-signup">No sign up required</span>`;


  const timeRow = e.Time ? `

    <div class="event-meta-row">

      <svg width="13" height="13" viewBox="0 0 24 24" fill="none" stroke="currentColor" stroke-width="2"><circle cx="12" cy="12" r="10"/><path d="M12 6v6l4 2"/></svg>

      ${e.Time}

    </div>` : '';


  const locRow = e.Location ? `

    <div class="event-meta-row">

      <svg width="13" height="13" viewBox="0 0 24 24" fill="none" stroke="currentColor" stroke-width="2"><path d="M21 10c0 7-9 13-9 13s-9-6-9-13a9 9 0 0118 0z"/><circle cx="12" cy="10" r="3"/></svg>

      ${e.Location}

    </div>` : '';


  return `<div class="event-card" style="animation-delay:${i * 0.05}s;opacity:${isPast ? '0.65' : '1'}">

    <div class="event-img">

      ${imgSection}

      <div class="event-img-overlay"></div>

      <div class="event-type-badge ${tc.badge}">${tc.label}</div>

      <div class="event-name">${e.Name}</div>

        ${timeRow}

        ${locRow}

      </div>

      ${e.Description ? `<div class="event-desc">${e.Description}</div>` : ''}

    </div>

  </div>`;

}
