// ── CSV parser ──

function parseCSV(text) {

  const lines = text.trim().split('\n');

  const headers = lines[0].split(',').map(h => h.trim().replace(/^"|"$/g, ''));

  return lines.slice(1).map(line => {

    const vals = [];

    let cur = '', inQ = false;

    for (let i = 0; i < line.length; i++) {

      const ch = line[i];

      if (ch === '"') { inQ = !inQ; }

      else if (ch === ',' && !inQ) { vals.push(cur.trim()); cur = ''; }

      else { cur += ch; }

    }

    vals.push(cur.trim());

    const obj = {};

    headers.forEach((h, i) => obj[h] = (vals[i] || '').replace(/^"|"$/g, ''));

    return obj;

  }).filter(r => Object.values(r).some(v => v));

}


// ── Tab switching ──

function showTab(tabId) {

  document.querySelectorAll('.tab-btn').forEach(b => b.classList.remove('active'));

  document.querySelectorAll('.section').forEach(s => s.classList.remove('active'));

  document.querySelector('[data-tab="' + tabId + '"]').classList.add('active');

  document.getElementById('tab-' + tabId).classList.add('active');

}


// ── Copy dua ──

function copyDua(btn) {

  const card = btn.closest('.dua-card');

  const arabic = card.querySelector('.dua-arabic').textContent.trim();

  const trans  = card.querySelector('.dua-transliteration').textContent.trim();

  const meaning = card.querySelector('.dua-translation').textContent.trim();

  const source = card.querySelector('.dua-source').textContent.trim();

  navigator.clipboard.writeText(arabic + '\n\n' + trans + '\n\n' + meaning + '\n\n' + source).then(() => {

    btn.textContent = 'Copied!';

    btn.classList.add('copied');

    setTimeout(() => { btn.textContent = 'Copy'; btn.classList.remove('copied'); }, 2000);

  });

}


// ── Mosque map ──

function showMosque(el) {

  document.querySelectorAll('.mosque-item').forEach(m => m.classList.remove('active'));

  el.classList.add('active');

  const q = encodeURIComponent(el.dataset.q);

  document.getElementById('mosque-map').src =

    'https://www.google.com/maps/embed/v1/search?key=' + MAPS_KEY + '&q=' + q + '&zoom=15';

}


// ── Render duas ──

async function loadDuas() {

  try {

    const res = await fetch(DUAS_URL);

    const rows = parseCSV(await res.text());

    const grid = document.getElementById('duas-grid');

    grid.innerHTML = rows.map(r => `

      <div class="dua-card">

        <div class="dua-label">${r.Label || ''}</div>

        <button class="copy-btn" onclick="copyDua(this)">Copy</button>

        <div class="dua-arabic">${r.Arabic || ''}</div>

        <div class="dua-transliteration">${r.Transliteration || ''}</div>

        <div class="dua-translation">"${r.Translation || ''}"</div>

        <div class="dua-source">${r.Source || ''}</div>

      </div>`).join('');

  } catch(e) {

    document.getElementById('duas-grid').innerHTML = '<p class="loading-msg">Could not load duas.</p>';

  }

}


// ── Render mosques ──

async function loadMosques() {

  try {

    const res = await fetch(MOSQUES_URL);

    const rows = parseCSV(await res.text());

    const list = document.getElementById('mosque-list');

    list.innerHTML = rows.map((r, i) => {

      const tags = (r.Tags||'').split(',').map(t => t.trim()).filter(Boolean);

      const tagHTML = tags.map(t => {

        let cls = '';

        if (t.toLowerCase().includes('jum')) cls = 'jummah';

        else if (t.toLowerCase().includes('wudu')) cls = 'wudu';

        else if (t.toLowerCase().includes('sister')) cls = 'sisters';

        return `<span class="mosque-tag ${cls}">${t}</span>`;

      }).join('');

      return `

      <div class="mosque-item ${i===0?'active':''}" data-q="${r.MapsQuery||''}" onclick="showMosque(this)">

        <div class="mosque-name">${r.Name || ''}</div>

        <div class="mosque-addr">${r.Address || ''}</div>

        ${tagHTML ? `<div class="mosque-tags">${tagHTML}</div>` : ''}

        <a class="mosque-dir-btn" href="https://maps.google.com/?q=${encodeURIComponent(r.Address||'')}" target="_blank">↗ Get Directions</a>

      </div>`;

    }).join('');

    // Set initial map to first mosque

    if (rows.length > 0) {

      const q = encodeURIComponent(rows[0].MapsQuery || rows[0].Name);

      document.getElementById('mosque-map').src =

        'https://www.google.com/maps/embed/v1/search?key=' + MAPS_KEY + '&q=' + q + '&zoom=15';

    }

  } catch(e) {

    document.getElementById('mosque-list').innerHTML = '<p class="loading-msg">Could not load locations.</p>';

  }

}
