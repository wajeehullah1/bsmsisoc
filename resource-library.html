// ── Drive API ─────────────────────────────────────────────────

async function fetchFolderFiles(folderId, year, phase) {

  let files = [];

  let pageToken = null;

  do {

    let url = `https://www.googleapis.com/drive/v3/files?q='${folderId}'+in+parents+and+trashed=false&fields=nextPageToken,files(id,name,mimeType)&key=${API_KEY}&pageSize=200`;

    if (pageToken) url += `&pageToken=${pageToken}`;

    const res = await fetch(url);

    const json = await res.json();

    if (json.error) throw new Error(json.error.message);

    for (const f of (json.files || [])) {

      if (f.mimeType === 'application/vnd.google-apps.folder') {

        // Recurse into subfolders, keeping same year/phase

        const sub = await fetchFolderFiles(f.id, year, phase);

        files = files.concat(sub);

      } else {

        const ext = f.name.split('.').pop().toLowerCase();

        const type = EXT_TYPE[ext] || 'doc';

        // Derive subject from filename prefix (e.g. "203 Theme 1..." → "Module 203")

        const moduleMatch = f.name.match(/^(\d{3})\s/);

        const subject = moduleMatch ? `Module ${moduleMatch[1]}` : guessSubject(f.name);

        files.push({ id: f.id, name: f.name.replace(/\.[^.]+$/, ''), subject, year, phase, type, driveId: f.id });

      }

    }

    pageToken = json.nextPageToken || null;

  } while (pageToken);

  return files;

}


function guessSubject(name) {

  const n = name.toLowerCase();

  if (n.includes('osce') || n.includes('history') || n.includes('clinical')) return 'OSCE';

  if (n.includes('anatomy')) return 'Anatomy';

  if (n.includes('pathology')) return 'Pathology';

  if (n.includes('pharmacology') || n.includes('drug')) return 'Pharmacology';

  if (n.includes('haem')) return 'Haematology';

  if (n.includes('cardio') || n.includes('cvs')) return 'Cardiology';

  if (n.includes('respiratory') || n.includes('lung') || n.includes('resp')) return 'Respiratory';

  if (n.includes('psychiatry') || n.includes('psych')) return 'Psychiatry';

  if (n.includes('mock') || n.includes('exam') || n.includes('question') || n.includes('fma')) return 'Mock Exams';

  if (n.includes('lecture') || n.includes('tracker')) return 'Trackers';

  if (n.includes('neurol')) return 'Neurology';

  if (n.includes('surgery')) return 'Surgery';

  return 'General';

}

          <div class="file-icon ${r.type}">${FILE_ICONS[r.type] || '📄'}</div>

          <div class="card-name">${r.name}</div>

        </div>

        <div class="card-meta">

          <span class="subject-tag">${r.subject}</span>

          <span class="open-link">Open ↗</span>

        </div>

      </div>`;

    });

    html += `</div></div>`;

  });

  area.innerHTML = html;

  area.querySelectorAll('.resource-card').forEach(card => {

    card.addEventListener('click', () => {

      window.open(`https://drive.google.com/file/d/${card.dataset.id}/view`, '_blank');

    });

  });

}
