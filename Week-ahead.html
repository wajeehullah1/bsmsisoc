<script>

const SHEET_CSV_URL = 'https://docs.google.com/spreadsheets/d/e/2PACX-1vTBxTrifZz0l8lD_8gFdk6rTNM91liCO_hU5rfolnx-Qbi90flWUcSBaJjqMbfwtlkEQ3iNFUrjJNVB/pub?output=csv';


const DAYS = ['Sunday','Monday','Tuesday','Wednesday','Thursday','Friday','Saturday'];

const MONTHS = ['Jan','Feb','Mar','Apr','May','Jun','Jul','Aug','Sep','Oct','Nov','Dec'];

const TYPE_LABELS = {

  talk:'Talk / Lecture', lecture:'Talk / Lecture', social:'Social',

  sports:'Sports', sisters:'Sisters Only', brothers:'Brothers Only',

  charity:'Charity', academic:'Academic'

};


function parseCSV(text) {

  const lines = text.trim().split('\n');

  if (lines.length < 2) return [];

  const headers = lines[0].split(',').map(h => h.replace(/^"|"$/g, '').trim());

  return lines.slice(1).map(line => {

    const fields = [];

    let cur = '', inQ = false;

    for (let i = 0; i < line.length; i++) {

      if (line[i] === '"') { inQ = !inQ; }

      else if (line[i] === ',' && !inQ) { fields.push(cur.trim()); cur = ''; }

      else { cur += line[i]; }

    }

    fields.push(cur.trim());

    const obj = {};

    headers.forEach((h, i) => obj[h] = (fields[i] || '').replace(/^"|"$/g, '').trim());

    return obj;

  }).filter(r => r.Name && r.Name.length > 0);

}


function parseDate(str) {

  if (!str) return null;

  // Handle M/D/YYYY or MM/DD/YYYY (Google Sheets default)

  const parts = str.split('/');

  if (parts.length === 3) {

    const [m, d, y] = parts;

    const date = new Date(parseInt(y), parseInt(m) - 1, parseInt(d));

    if (!isNaN(date)) return date;

  }

  // Fallback: YYYY-MM-DD or other formats

  const d = new Date(str);

  if (!isNaN(d)) return d;

  return null;

}


function getWeekBounds() {

  const today = new Date(); today.setHours(0,0,0,0);

  const end = new Date(today); end.setDate(today.getDate() + 7);

  return { today, end };

}


function buildCard(e, i) {

  const date = parseDate(e.Date);

  // Support multiple types e.g. "Social, Sisters" — use first recognised one for styling

  const types = (e.Type || '').toLowerCase().split(',').map(t => t.trim());

  const typeKey = types.find(t => TYPE_LABELS[t]) || types[0] || 'event';

  const typeLabels = types.map(t => TYPE_LABELS[t] || e.Type || 'Event').filter((v,i,a) => a.indexOf(v)===i);


  const dayLabel = date

    ? `${DAYS[date.getDay()]} ${date.getDate()} ${MONTHS[date.getMonth()]}`

    : 'Coming up';


  const timeRow = e.Time

    ? `<div class="card-time">

        <svg width="11" height="11" viewBox="0 0 24 24" fill="none" stroke="currentColor" stroke-width="2"><circle cx="12" cy="12" r="10"/><path d="M12 6v6l4 2"/></svg>

        ${e.Time}

      </div>` : '';


  const locRow = e.Location

    ? `<div class="card-location">

        <svg width="11" height="11" viewBox="0 0 24 24" fill="none" stroke="currentColor" stroke-width="2"><path d="M21 10c0 7-9 13-9 13s-9-6-9-13a9 9 0 0118 0z"/><circle cx="12" cy="10" r="3"/></svg>

        ${e.Location}

      </div>` : '';


  const signup = e.FormLink

    ? `<a class="card-signup" href="${e.FormLink}" target="_blank">Sign Up ↗</a>` : '';


  return `<div class="event-card type-${typeKey}" style="animation-delay:${i*0.06}s">

    <div class="card-day">${dayLabel}</div>

    <div class="card-name">${e.Name}</div>

    ${timeRow}${locRow}

    ${typeLabels.map(l => `<span class="card-type-badge">${l}</span>`).join(' ')}

    ${signup}

  </div>`;

}


async function init() {

  const { today, end } = getWeekBounds();


  // Date range label

  document.getElementById('date-range').textContent =

    `${DAYS[today.getDay()]} ${today.getDate()} ${MONTHS[today.getMonth()]} – ${DAYS[end.getDay()]} ${end.getDate()} ${MONTHS[end.getMonth()]}`;


  try {

    const res = await fetch(SHEET_CSV_URL);

    const text = await res.text();

    const all = parseCSV(text);


    // Filter to this week + undated events

    const week = all.filter(e => {

      const d = parseDate(e.Date);

      if (!d) return false; // only show dated events in week view

      return d >= today && d <= end;

    }).sort((a, b) => parseDate(a.Date) - parseDate(b.Date));


    const container = document.getElementById('cards-container');


    if (week.length === 0) {

      container.innerHTML = `<div class="empty-state">

        <div class="moon">🌙</div>

        <p>No events this week — check back soon insha'Allah.<br>

        <a href="#" style="color:var(--green-light);font-weight:500">See all upcoming events ↗</a></p>

      </div>`;

      return;

    }


    container.innerHTML = `<div class="cards-row">${week.map((e,i) => buildCard(e,i)).join('')}</div>`;


  } catch(err) {

    document.getElementById('cards-container').innerHTML = `<div class="empty-state">

      <p>Could not load events right now.</p>

    </div>`;

  }

}


init();

</script>
