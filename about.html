function parseCSV(text) {

  const lines = text.trim().split('\n');

  if (lines.length < 2) return [];

  const headers = lines[0].split(',').map(h => h.replace(/^"|"$/g, '').trim());

  return lines.slice(1).map(line => {

    const fields = [];

    let cur = '', inQ = false;

    for (let i = 0; i < line.length; i++) {

      if (line[i] === '"') { inQ = !inQ; }

      else if (line[i] === ',' && !inQ) { fields.push(cur.trim()); cur = ''; }

      else { cur += line[i]; }

    }

    fields.push(cur.trim());

    const obj = {};

    headers.forEach((h, i) => obj[h] = (fields[i] || '').replace(/^"|"$/g, '').trim());

    return obj;

  }).filter(r => r.Name && r.Name.length > 0);

}


function normaliseDriveUrl(url) {

  if (!url) return null;

  // Handle /file/d/FILE_ID/view format

  const fileMatch = url.match(/\/file\/d\/([a-zA-Z0-9_-]+)/);

  if (fileMatch) return `https://drive.google.com/thumbnail?id=${fileMatch[1]}&sz=w400`;

  // Handle ?id=FILE_ID format (already thumbnail style)

  const idMatch = url.match(/[?&]id=([a-zA-Z0-9_-]+)/);

  if (idMatch) return `https://drive.google.com/thumbnail?id=${idMatch[1]}&sz=w400`;

  // Return as-is if neither pattern matches

  return url;

}

  const tagClass  = isSister ? 'tag-sister' : 'tag-brother';

  const tagLabel  = isSister ? 'Sister' : 'Brother';

  const delay     = Math.min(i * 0.04, 0.4);


  return `<div class="member-card" style="animation-delay:${delay}s">

    <img class="member-avatar" src="${avatarSrc}" alt="${member.Name}"

      onerror="this.style.display='none';this.nextElementSibling.style.display='flex'">

      <span class="member-gender-tag ${tagClass}">${tagLabel}</span>

    </div>

  </div>`;

}


function render(data) {

  // Group by section

  const sections = {};

  data.forEach(m => {

    const sec = m.Section || 'Other';

    if (!sections[sec]) sections[sec] = [];

    sections[sec].push(m);

  });


  // Build ordered list of sections

  const orderedSections = [

    ...SECTION_ORDER.filter(s => sections[s]),

    ...Object.keys(sections).filter(s => !SECTION_ORDER.includes(s))

  ];


  const main = document.getElementById('main');

  main.innerHTML = orderedSections.map(secName => {

    const members = sections[secName];

    const isHead  = secName === 'Head';

    const cards   = members.map((m, i) => buildCard(m, i)).join('');

    return `<div class="committee-section">

      <div class="section-heading"><h2>${secName}</h2></div>

      <div class="cards-grid${isHead ? ' head' : ''}">${cards}</div>

    </div>`;

  }).join('');

}
