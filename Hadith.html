<script>

const SHEET_URL = 'https://docs.google.com/spreadsheets/d/e/2PACX-1vR4lLkQEHC_s2OdYYi-Tpwwg5ZO--kf3oM26LrBrp9eARrv6T2XovmF1QmdSLsEs2Z096ziL26mJqKy/pub?output=csv';

const INTERVAL = 6000;

let current = 0;

let timer, progressTimer, progressStart;

let hadiths = [];


const track    = document.getElementById('track');

const dotsEl   = document.getElementById('dots');

const progressEl = document.getElementById('progress');


function parseCSV(text) {

  const lines = text.trim().split('\n');

  const headers = lines[0].split(',').map(h => h.trim().replace(/^"|"$/g, ''));

  return lines.slice(1).map(line => {

    const vals = [];

    let cur = '', inQ = false;

    for (let i = 0; i < line.length; i++) {

      const ch = line[i];

      if (ch === '"') { inQ = !inQ; }

      else if (ch === ',' && !inQ) { vals.push(cur); cur = ''; }

      else { cur += ch; }

    }

    vals.push(cur);

    const obj = {};

    headers.forEach((h, i) => obj[h] = (vals[i] || '').replace(/^"|"$/g, '').trim());

    return obj;

  }).filter(r => r.Arabic || r.Text);

}


function buildCarousel(data) {

  hadiths = data;

  track.innerHTML = '';

  dotsEl.innerHTML = '';


  hadiths.forEach(h => {

    const card = document.createElement('div');

    card.className = 'hadith-card';

    card.innerHTML = `

      <div class="quote-mark">"</div>

      <p class="hadith-arabic">${h.Arabic || ''}</p>

      <div class="hadith-divider"></div>

      <p class="hadith-text">"${h.Text || ''}"</p>

      <p class="hadith-source">${h.Source || ''}</p>

    `;

    track.appendChild(card);

  });


  hadiths.forEach((_, i) => {

    const dot = document.createElement('button');

    dot.className = 'dot' + (i === 0 ? ' active' : '');

    dot.addEventListener('click', () => goTo(i));

    dotsEl.appendChild(dot);

  });


  goTo(0);

  startAuto();

}


function goTo(idx) {

  current = (idx + hadiths.length) % hadiths.length;

  track.style.transform = `translateX(-${current * 100}%)`;

  document.querySelectorAll('.dot').forEach((d, i) => d.classList.toggle('active', i === current));

  resetProgress();

}


function resetProgress() {

  clearInterval(progressTimer);

  progressEl.style.width = '0%';

  progressStart = Date.now();

  progressTimer = setInterval(() => {

    const pct = Math.min(((Date.now() - progressStart) / INTERVAL) * 100, 100);

    progressEl.style.width = pct + '%';

  }, 50);

}


function startAuto() {

  clearInterval(timer);

  timer = setInterval(() => goTo(current + 1), INTERVAL);

}


document.querySelector('.prev').addEventListener('click', () => { goTo(current - 1); clearInterval(timer); startAuto(); });

document.querySelector('.next').addEventListener('click', () => { goTo(current + 1); clearInterval(timer); startAuto(); });


const outer = document.querySelector('.carousel-outer');

outer.addEventListener('mouseenter', () => { clearInterval(timer); clearInterval(progressTimer); });

outer.addEventListener('mouseleave', () => { startAuto(); resetProgress(); });


let touchStartX = 0;

track.addEventListener('touchstart', e => { touchStartX = e.touches[0].clientX; }, { passive: true });

track.addEventListener('touchend', e => {

  const diff = touchStartX - e.changedTouches[0].clientX;

  if (Math.abs(diff) > 40) { goTo(current + (diff > 0 ? 1 : -1)); clearInterval(timer); startAuto(); }

});


// Load from sheet

fetch(SHEET_URL)

  .then(r => r.text())

  .then(text => buildCarousel(parseCSV(text)))

  .catch(() => {

    // Fallback: show a single card if sheet fails

    track.innerHTML = '<div class="hadith-card"><p class="hadith-text">Could not load hadiths. Please check your internet connection.</p></div>';

  });

</script>
